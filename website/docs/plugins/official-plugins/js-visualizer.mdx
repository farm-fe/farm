# @farmfe/js-plugin-visualizer

Visualize Farm bundle size and module composition.

## Installation

<Tabs>
  <TabItem value="npm" label="npm">
    <CodeBlock>npm install @farmfe/js-plugin-visualizer</CodeBlock>
  </TabItem>
  <TabItem value="yarn" label="yarn">
    <CodeBlock>yarn add @farmfe/js-plugin-visualizer</CodeBlock>
  </TabItem>
  <TabItem value="pnpm" label="pnpm">
    <CodeBlock>pnpm add @farmfe/js-plugin-visualizer</CodeBlock>
  </TabItem>
</Tabs>

## Usage

```ts {2,11}
import { defineConfig } from '@farmfe/core';
import visualizer from '@farmfe/js-plugin-visualizer';

export default defineConfig({
  compilation: {
    input: {
      index: './index.html'
    }
  },
  plugins: [
    visualizer()
  ]
});
```

## Options

```ts
export interface RecordViewerOptions {
  /**
   * Specify hostname.
   * @default '127.0.0.1'
   */
  host?: string;

  /**
   * Specify port.
   * @default 9527
   */
  port?: number;
}
```

### `host`

Host for the visualizer server. Defaults to `127.0.0.1`.

### `port`

Port for the visualizer server. Defaults to `9527`.
